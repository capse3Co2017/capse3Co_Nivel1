321 char* itoa(int value, char* result, int base)
322 {
323 // check that the base if valid
324 if (base < 2 || base > 36) { *result = '\0'; return result; }
325 
326 char* ptr = result, *ptr1 = result, tmp_char;
327 int tmp_value;
328 
329 do {
330 tmp_value = value;
331 value /= base;
332 *ptr++ = "zyxwvutsrqponmlkjihgfedcba9876543210123456789abcdefghijklmnopqrstuvwxyz" [35 + (tmp_value - value * base)];
333 } while ( value );
334 
335 // Apply negative sign
336 if (tmp_value < 0) *ptr++ = '-';
337 *ptr-- = '\0';
338 while(ptr1 < ptr) {
339 tmp_char = *ptr;
340 *ptr--= *ptr1;
341 *ptr1++ = tmp_char;
342 }
343 return result;
344 }

643 int16_t vectorR2ToAngle(int16_t axis1, int16_t axis2)
644 {
645 float angle = 0;
646 float heading = 0;
647 
648 // Angulo en radianes
649 heading = atan2((float)axis2, (float)axis1);
650 
651 // Angulo en radianes ajustado
652 if (heading < 0.0)
653  heading += 2.0 * 3.1415;
654 
655 // Angulo en grados
656 angle = (heading * 180.0 / 3.1415); //M_PI
657 
658 return (int16_t)angle;
659 }
