61 /* Configure keypad pins */
62 bool_t keypadConfig( keypad_t* keypad,
63  gpioMap_t* keypadRowPins, uint8_t keypadRowSize,
64  gpioMap_t* keypadColPins, uint8_t keypadColSize ){
65 
66  bool_t retVal = TRUE;
67 
68  uint8_t i = 0;
69 
70  // Check if values are not invalid
71  if( keypadRowPins == NULL || keypadColPins == NULL ||
72  keypadRowSize <= 0 || keypadColSize <= 0 ){
73  retVal = FALSE;
74  }
75 
76  // Configure keypad instance
77  keypad->keypadRowPins = keypadRowPins;
78  keypad->keypadRowSize = keypadRowSize;
79  keypad->keypadColPins = keypadColPins;
80  keypad->keypadColSize = keypadColSize;
81 
82  // Configure Rows as Outputs
83  for( i=0; i<keypadRowSize; i++ ){
84  gpioConfig( keypad->keypadRowPins[i], GPIO_OUTPUT );
85  }
86 
87  // Configure Columns as Inputs with pull-up resistors enable
88  for( i=0; i<keypadColSize; i++ ){
89  gpioConfig( keypad->keypadColPins[i], GPIO_INPUT_PULLUP );
90  }
91 
92  return retVal;
93 }

96 /* Return TRUE if any key is pressed or FALSE (0) in other cases.
97  * If exist key pressed write pressed key on key variable */
98 bool_t keypadRead( keypad_t* keypad, uint16_t* key ){
99 
100  bool_t retVal = FALSE;
101 
102  uint8_t r = 0; // Rows
103  uint8_t c = 0; // Columns
104 
105  // Put all Rows in LOW state
106  for( r=0; r<keypad->keypadRowSize; r++ ){
107  gpioWrite( keypad->keypadRowPins[r], LOW );
108  }
109 
110  // Check all Columns to search if any key is pressed
111  for( c=0; c<keypad->keypadColSize; c++ ){
112 
113  // If reads a LOW state in a column then that key may be pressed
114  if( !gpioRead( keypad->keypadColPins[c] ) ){
115 
116  delay( 50 ); // Debounce 50 ms
117 
118  // Put all Rows in HIGH state except first one
119  for( r=1; r<keypad->keypadRowSize; r++ ){
120  gpioWrite( keypad->keypadRowPins[r], HIGH );
121  }
122 
123  // Search what key are pressed
124  for( r=0; r<keypad->keypadRowSize; r++ ){
125 
126  // Put the Row[r-1] in HIGH state and the Row[r] in LOW state
127  if( r>0 ){ // Prevents negative index in array
128  gpioWrite( keypad->keypadRowPins[r-1], HIGH );
129  }
130  gpioWrite( keypad->keypadRowPins[r], LOW );
131 
132  // Check Columns[c] at Row[r] to search if the key is pressed
133  // if that key is pressed (LOW state) then retuns the key
134  if( !gpioRead( keypad->keypadColPins[c] ) ){
135  *key = (uint16_t)r * (uint16_t)(keypad->keypadColSize) + (uint16_t)c;
136  retVal = TRUE;
137  return retVal;
138  }
139  }
140 
141  }
142  }
143 
144  /*
145  4 rows * 5 columns Keypad
146 
147  c0 c1 c2 c3 c4
148  r0 0 1 2 3 4
149  r1 5 6 7 8 9 Press r[i] c[j] => (i) * amountOfColumns + (j)
150  r2 10 11 12 13 14
151  r3 15 16 17 18 19
152  */
153 
154  // if no key are pressed then retun FALSE
155  return retVal;
156 }
