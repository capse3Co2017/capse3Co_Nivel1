59  uartMap_t uart, waitForReceiveStringOrTimeout_t* instance ){

61  uint8_t receiveByte;

69  instance->stringIndex = 0;
70 
71  instance->state = UART_RECEIVE_STRING_RECEIVING;
72 
73  break;
74 
75  case UART_RECEIVE_STRING_RECEIVING:
76

78 
79  // TODO: DEBUG

90  }
91 
92  }
93 
94  if( delayRead( &(instance->delay) ) ){

97

99

101  instance->state = UART_RECEIVE_STRING_CONFIG;
102  break;
103 
104  case UART_RECEIVE_STRING_TIMEOUT:
105  instance->state = UART_RECEIVE_STRING_CONFIG;
106  break;

118 bool_t waitForReceiveStringOrTimeoutBlocking(
119  uartMap_t uart, char* string, uint16_t stringSize, tick_t timeout ){
120 
121  bool_t retVal = TRUE; // True if OK
122 
123  waitForReceiveStringOrTimeout_t waitText;
124  waitForReceiveStringOrTimeoutState_t waitTextState;
125 
126  waitTextState = UART_RECEIVE_STRING_CONFIG;
127 
128  waitText.state = UART_RECEIVE_STRING_CONFIG;
129  waitText.string = string;
130  waitText.stringSize = stringSize;
131  waitText.timeout = timeout;
132 
133  while( waitTextState != UART_RECEIVE_STRING_RECEIVED_OK &&
134  waitTextState != UART_RECEIVE_STRING_TIMEOUT ){
135  waitTextState = waitForReceiveStringOrTimeout( uart, &waitText );
136  }
137 
138  if( waitTextState == UART_RECEIVE_STRING_TIMEOUT ){
139  retVal = FALSE;
140  }
141 
142  return retVal;
143 }
