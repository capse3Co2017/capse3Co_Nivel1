36 /*The SCT (State Configurable Timer) is a feature included in some of LPC's microcontrollers
37  * that provides a high resolution PWM (or just another timer).
38  * It's like a normal timer but with multiple Compare Match values (16),
39  * and can be therefore used to generate several PWM signals with THE SAME PERIOD
40  * For more information about the STCPWM peripheral, refer to the Chapter 39 of
41  * the LPC43xx user manual

100 /*
101  * @brief: Initialize the SCT peripheral with the given frequency
102  * @param: frequency: value in Hz
103  * @note: there can only be 1 frequency in all the SCT peripheral.
104  */
105 void Sct_Init(uint32_t frequency)
106 {
107  /* Source: https://www.lpcware.com/content/faq/how-use-sct-standard-pwm-using-lpcopen */
108  /* Initialize the SCT as PWM and set frequency */
109  Chip_SCTPWM_Init(LPC_SCT);
110  Chip_SCTPWM_SetRate(LPC_SCT, frequency);
111 
112  Chip_SCTPWM_Start(LPC_SCT);
113 }

115 /*
116  * @brief Enables pwm function for the given pin
117  * @param sctNumber: pin where the pwm signal will be generated
118  */
119 void Sct_EnablePwmFor(uint8_t sctNumber)
120 {
121  /*Enable SCT function on pin*/
122  Chip_SCU_PinMux(SCTdataList[sctNumber].port , SCTdataList[sctNumber].pin , SCU_MODE_INACT , CTOUT_FUNC);
123  /*Sets pin as PWM output and gives it an index (SCTdataList[sctNumber].mode+1)*/
124  Chip_SCTPWM_SetOutPin(LPC_SCT, sctNumber+1, sctNumber);
125 
126  /* Start with 0% duty cycle */
127  Sct_SetDutyCycle(sctNumber, Chip_SCTPWM_PercentageToTicks(LPC_SCT,0));
128 }
