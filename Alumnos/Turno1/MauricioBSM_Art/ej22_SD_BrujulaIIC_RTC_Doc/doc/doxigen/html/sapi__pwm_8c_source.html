85 /* Enter a pwm number, get a sct number
86  * Since this module works with pwm numbers, but uses sct channels to generate
87  * the signal, its necessary to connect pwm number with the SctMap_t (sAPI_PeripheralMap.h).
88  * This way the user sets "pwms", while using the sct peripheral internally*/
89 static const uint8_t pwmMap[PWM_TOTALNUMBER] = {
90  /* PWM0 */ CTOUT1, /* T_FIL1 */
91  /* PWM1 */ CTOUT12, /* T_COL2 */
92  /* PWM2 */ CTOUT10, /* T_COL0 */
93  /* PWM3 */ CTOUT0, /* T_FIL2 */
94  /* PWM4 */ CTOUT3, /* T_FIL3 */
95  /* PWM5 */ CTOUT13, /* T_COL1 */

130 /*
131  * @brief: adds pwm to the the list of working pwms
132  * @param: pwmNumber: ID of the pwm, from 0 to 10
133  * @return: True if pwm was successfully attached, False if not.
134  */
135 static bool_t pwmAttach( pwmMap_t pwmNumber){
136 
137  bool_t success = FALSE;
138  uint8_t position = 0;
139 
140  position = pwmIsAttached(pwmNumber);
141  if(position==0){
142  position = pwmIsAttached(EMPTY_POSITION); /* Searches for the first empty position */
143  if(position){ /* if position==0 => there is no room in the list for another pwm */
144  AttachedPWMList[position-1] = pwmNumber;
145  Sct_EnablePwmFor(pwmMap[pwmNumber]);
146  success = TRUE;
147  }
148  }
149  return success;
150 }

221 bool_t pwmConfig( pwmMap_t pwmNumber, pwmConfig_t config){
222 
223  bool_t ret_val = 1;
224 
225  switch(config){

242 
243  default:
244  ret_val = 0;
245  break;
246  }
247 
248  return ret_val;

251 /*
252  * @brief: Tells if the pwm is currently active, and its position
253  * @param: pwmNumber: ID of the pwm, from 0 to 10
254  * @return: position (1 ~ PWM_TOTALNUMBER), 0 if the element was not found.
255  */
256 uint8_t pwmIsAttached( pwmMap_t pwmNumber )
257 {
258  uint8_t position = 0, positionInList = 0;
259  while ( (position < PWM_TOTALNUMBER) &&
260  (pwmNumber != AttachedPWMList[position]) ) {
261  position++;
262  }
263 
264  if (position < PWM_TOTALNUMBER){
265  positionInList = position + 1;
266  } else{
267  positionInList = 0;
268  }
269 
270  return positionInList;
271 }
