94 bool_t delayRead( delay_t * delay ){
95 
96  bool_t timeArrived = 0;
97 
98  if( !delay->running ){
99  delay->startTime = tickRead();
100  delay->running = 1;
101  }
102  else{
103  if ( (tick_t)(tickRead() - delay->startTime) >= delay->duration ){
104  timeArrived = 1;
