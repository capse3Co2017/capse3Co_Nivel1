207 bool_t gpioConfig( gpioMap_t pin, gpioConfig_t config ){
208 
209  bool_t ret_val = 1;
210 
211  int8_t pinNamePort = 0;
212  int8_t pinNamePin = 0;
213 
214  int8_t func = 0;
215 
216  int8_t gpioPort = 0;
217  int8_t gpioPin = 0;
218 
219  gpioObtainPinConfig( pin, &pinNamePort, &pinNamePin, &func,
220  &gpioPort, &gpioPin );
221 
222  switch(config){
223 
224  case GPIO_ENABLE:
225  /* Initializes GPIO */
226  Chip_GPIO_Init(LPC_GPIO_PORT);
227  break;
228 
229  case GPIO_INPUT:
230  Chip_SCU_PinMux(
231  pinNamePort,
232  pinNamePin,
233  SCU_MODE_INACT | SCU_MODE_INBUFF_EN | SCU_MODE_ZIF_DIS,
234  func
235  );
236  Chip_GPIO_SetDir( LPC_GPIO_PORT, gpioPort, ( 1 << gpioPin ), GPIO_INPUT );
237  break;
238 
239  case GPIO_INPUT_PULLUP:
240  Chip_SCU_PinMux(
241  pinNamePort,
242  pinNamePin,
243  SCU_MODE_PULLUP | SCU_MODE_INBUFF_EN | SCU_MODE_ZIF_DIS,
244  func
245  );
246  Chip_GPIO_SetDir( LPC_GPIO_PORT, gpioPort, ( 1 << gpioPin ), GPIO_INPUT );
247  break;
248 
249  case GPIO_INPUT_PULLDOWN:
250  Chip_SCU_PinMux(
251  pinNamePort,
252  pinNamePin,
253  SCU_MODE_PULLDOWN | SCU_MODE_INBUFF_EN | SCU_MODE_ZIF_DIS,
254  func
255  );
256  Chip_GPIO_SetDir( LPC_GPIO_PORT, gpioPort, ( 1 << gpioPin ), GPIO_INPUT );
257  break;
258  case GPIO_INPUT_PULLUP_PULLDOWN:
259  Chip_SCU_PinMux(
260  pinNamePort,
261  pinNamePin,
262  SCU_MODE_REPEATER | SCU_MODE_INBUFF_EN | SCU_MODE_ZIF_DIS,
263  func
264  );
265  Chip_GPIO_SetDir( LPC_GPIO_PORT, gpioPort, ( 1 << gpioPin ), GPIO_INPUT );
266  break;
267 
268  case GPIO_OUTPUT:
269  Chip_SCU_PinMux(
270  pinNamePort,
271  pinNamePin,
272  SCU_MODE_INACT | SCU_MODE_ZIF_DIS | SCU_MODE_INBUFF_EN,
273  func
274  );
275  Chip_GPIO_SetDir( LPC_GPIO_PORT, gpioPort, ( 1 << gpioPin ), GPIO_OUTPUT );
276  Chip_GPIO_SetPinState( LPC_GPIO_PORT, gpioPort, gpioPin, 0);
277  break;
278 
279  default:
280  ret_val = 0;
281  break;
282  }
283 
284  return ret_val;
285 
286 }
