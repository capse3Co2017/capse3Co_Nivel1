110 /*
111  * @Brief Initialize Timer peripheral
112  * @param timerNumber: Timer number, 0 to 3
113  * @param ticks: Number of ticks required to finish the cycle.
114  * @param voidFunctionPointer: function to be executed at the end of the timer cycle
115  * @return nothing
116  * @note For the 'ticks' parameter, see function Timer_microsecondsToTicks
117  */
118 void Timer_Init( uint8_t timerNumber, uint32_t ticks,
119  voidFunctionPointer_t voidFunctionPointer){
120  /* Source:
121  http://docs.lpcware.com/lpcopen/v1.03/lpc18xx__43xx_2examples_2periph_2periph__blinky_2blinky_8c_source.html */
122 
123  /*If timer period = CompareMatch0 Period = 0 => ERROR*/
124  if (ticks==0){
125  errorOcurred();
126  }
127 
128  /* Enable timer clock and reset it */
129  Chip_TIMER_Init(timer_sd[timerNumber].name);
130  Chip_RGU_TriggerReset(timer_sd[timerNumber].RGU);
131  while (Chip_RGU_InReset(timer_sd[timerNumber].RGU)) {}
132  Chip_TIMER_Reset(timer_sd[timerNumber].name);
133 
134  /* Update the defalut function pointer name of the Compare match 0*/
135  timer_dd[timerNumber].timerCompareMatchFunctionPointer[TIMERCOMPAREMATCH0] = voidFunctionPointer;
136 
137  /* Initialize compare match with the specified ticks (number of counts needed to clear the match counter) */
138  Chip_TIMER_MatchEnableInt(timer_sd[timerNumber].name, TIMERCOMPAREMATCH0);
139  Chip_TIMER_SetMatch(timer_sd[timerNumber].name, TIMERCOMPAREMATCH0, ticks);
140 
141  /* Makes Timer Match 0 period the timer period*/
142  Chip_TIMER_ResetOnMatchEnable(timer_sd[timerNumber].name, TIMERCOMPAREMATCH0);
143 
144  /*Enable timer*/
145  Chip_TIMER_Enable(timer_sd[timerNumber].name);
146 
147  /* Enable timer interrupt */
148  NVIC_SetPriority(timer_sd[timerNumber].IRQn, MAX_SYSCALL_INTERRUPT_PRIORITY+1);
149  NVIC_EnableIRQ(timer_sd[timerNumber].IRQn);
150  NVIC_ClearPendingIRQ(timer_sd[timerNumber].IRQn);
151 }

174 /*
175  * @Brief Enables a compare match in a timer

177  * @param compareMatchNumber: Compare match number, 1 to 3
178  * @param ticks: Number of ticks required to reach the compare match.

183 void Timer_EnableCompareMatch( uint8_t timerNumber, uint8_t compareMatchNumber,
184  uint32_t ticks,
185  voidFunctionPointer_t voidFunctionPointer){
186 
187  timer_dd[timerNumber].timerCompareMatchFunctionPointer[compareMatchNumber] = voidFunctionPointer;
188 
189  Chip_TIMER_MatchEnableInt(timer_sd[timerNumber].name, compareMatchNumber);
190  Chip_TIMER_SetMatch(timer_sd[timerNumber].name, compareMatchNumber, ticks);
191 }

229 void TIMER0_IRQHandler(void){
230 
231  uint8_t compareMatchNumber = 0;
232 
233  for( compareMatchNumber = TIMERCOMPAREMATCH0;
234  compareMatchNumber <= TIMERCOMPAREMATCH3;
235  compareMatchNumber++ ){
236  if( Chip_TIMER_MatchPending(LPC_TIMER0, compareMatchNumber) ){
237  /*Run the functions saved in the timer dynamic data structure*/
238  (*timer_dd[TIMER0].timerCompareMatchFunctionPointer[compareMatchNumber])();
239  Chip_TIMER_ClearMatch(LPC_TIMER0, compareMatchNumber);
240  }
241  }
242 }
