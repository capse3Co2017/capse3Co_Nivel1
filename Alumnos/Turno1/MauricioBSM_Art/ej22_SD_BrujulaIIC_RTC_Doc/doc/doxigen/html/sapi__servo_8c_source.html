134 /*
135  * @brief Converts a value in value to micro seconds for a specific type of servo (see the defines)
136  * @param value: value of the servo, from 0 to 180
137  * @return Equivalent in microseconds for a specific type of servo (see the defines)
138  * @note Should be used with Timer_microsecondsToTicks to use some of
139  * the functions in the sAPI_Timer that requires ticks as a parameter
140 */
141 static uint32_t valueToMicroseconds(uint8_t value){
142 
143  return (SERVO_MINUPTIME_PERIOD+(value*SERVO_MAXUPTIME_PERIOD)/180);
144 }

146 /*
147  * @brief: compare match 0 function. The one that is executed when the cycle ends
148  * (to visualize it, think about the 'timer ramp')
149  * @note: this function can't be generalized because sAPI_Timer functions expect void-void function pointers
150  */
151 static void timer1CompareMatch0func(void)
152 {
153  uint8_t servoListPosition= 0;
154 
155  for(servoListPosition=0;servoListPosition<3;servoListPosition++)
156  {
157  if(AttachedServoList[servoListPosition].servo != EMPTY_POSITION)
158  {
159  gpioWrite(servoMap[AttachedServoList[servoListPosition].servo],TRUE);
160  Timer_SetCompareMatch( AttachedServoList[servoListPosition].associatedTimer,
161  AttachedServoList[servoListPosition].associatedCompareMatch,
162  Timer_microsecondsToTicks(valueToMicroseconds(AttachedServoList[servoListPosition].value)));
163  }
164  }
165 }

261 /*
262  * @brief: adds a servo to the active servos list
263  * @param servoNumber: ID of the servo, from 0 to 8
264  * @return: True if servo was successfully attached, False if not.
265  */
266 static bool_t servoAttach( servoMap_t servoNumber)
267 {
268  bool_t success = FALSE;
269  uint8_t position = 0;
270 
271  /* Pin must b config as Output */
272  gpioConfig( (gpioMap_t)servoNumber, GPIO_OUTPUT );
273 
274  position = servoIsAttached(servoNumber);
275  if(position==0)
276  {
277  position = servoIsAttached(EMPTY_POSITION); /* Searches for the first empty position */
278  if(position) /* if position==0 => there is no room in the list for another servo */
279  {
280  AttachedServoList[position-1].servo = servoNumber;
281  /* Enables the compare match interrupt */
282  Timer_EnableCompareMatch( AttachedServoList[position-1].associatedTimer,
283  AttachedServoList[position-1].associatedCompareMatch,
284  Timer_microsecondsToTicks(valueToMicroseconds(AttachedServoList[position-1].value)),
285  AttachedServoList[position-1].associatedFunction
286  );
287  success = TRUE;
288  }
289  }
290 
291  return success;
292 }

294 /*
295  * @brief: removes a servo from the active servos list
296  * @param servoNumber: ID of the servo, from 0 to 8
297  * @return: True if servo was successfully detached, False if not.
298  */
299 static bool_t servoDetach( servoMap_t servoNumber )
300 {
301  bool_t success = FALSE;
302  uint8_t position = 0;
303 
304  position = servoIsAttached(servoNumber);
305 
306  if(position)
307  {
308  AttachedServoList[position-1].servo = EMPTY_POSITION;
309  AttachedServoList[position-1].value = 0;
310  Timer_DisableCompareMatch( AttachedServoList[position-1].associatedTimer,
311  AttachedServoList[position-1].associatedCompareMatch);
312  success = TRUE;
313  }
314  return success;
315 }

319 /*
320  * @Brief: Initializes the servo peripheral
321  * @param uint8_t servoNumber
322  * @param uint8_t config
323  * @return nothing
324  * @IMPORTANT: this function uses Timer 1, 2 and 3 to generate the servo signals, so
325  * they won't be available to use.
326  */
327 bool_t servoConfig( servoMap_t servoNumber, servoConfig_t config ){
328 
329  bool_t ret_val = 1;
330 
331  switch(config){
332 
333  case SERVO_ENABLE:
334  servoInitTimers();
335  break;
336 
337  case SERVO_DISABLE:
338  ret_val = 0;
339  break;
340 
341  case SERVO_ENABLE_OUTPUT:
342  ret_val = servoAttach( servoNumber );
343  break;
344 
345  case SERVO_DISABLE_OUTPUT:
346  ret_val = servoDetach( servoNumber );
347  break;
348 
349  default:
350  ret_val = 0;
351  break;
352  }
353 
354  return ret_val;
355 }

357 /*
358  * @brief: Tells if the servo is currently active, and its position
359  * @param: servoNumber: ID of the servo, from 0 to 8
360  * @param: value: value of the servo, from 0 to 180
361  * @return: position (1 ~ SERVO_TOTALNUMBER), 0 if the element was not found.
362  */
363 uint8_t servoIsAttached( servoMap_t servoNumber ){
364 
365  uint8_t position = 0, positionInList = 0;
366  while ( (position < SERVO_TOTALNUMBER) &&
367  (servoNumber != AttachedServoList[position].servo) ){
368  position++;
369  }
370 
371  if (position < SERVO_TOTALNUMBER){
372  positionInList = position + 1;
373  } else{
374  positionInList = 0;
375  }
376 
377  return positionInList;
378 }

380 /*

382  * @param: servoNumber: ID of the servo, from 0 to 8
383  * @return: value of the servo (0 ~ 180).
384  * If an error ocurred, return = EMPTY_POSITION = 255

387
