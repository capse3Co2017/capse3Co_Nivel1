59 /* Tick rate configuration 1 to 50 ms */
60 bool_t tickConfig(tick_t tickRateMSvalue, sAPI_FuncPtr_t tickHook ) {
61 
62  bool_t ret_val = 1;
63  tick_t tickRateHz = 0;
64 
65  if( tickHook ){
66  tickHookFunction = tickHook;
67  }
68 
69  if( (tickRateMSvalue >= 1) && (tickRateMSvalue <= 50) ){
70 
71  tickRateMS = tickRateMSvalue;
72 
73  /*
74  tickRateHz = 1000 => 1000 ticks per second => 1 ms tick
75  tickRateHz = 200 => 200 ticks per second => 5 ms tick
76  tickRateHz = 100 => 100 ticks per second => 10 ms tick
77  tickRateHz = 20 => 20 ticks per second => 50 ms tick
78  */
79  tickRateHz = 1000 / tickRateMSvalue;
80 
81  /* Init SysTick interrupt, tickRateHz ticks per second */
82  SysTick_Config( SystemCoreClock / tickRateHz);
83  }
84  else{
85  /* Error, tickRateMS variable not in range (1 <= tickRateMS <= 50) */
86  ret_val = 0;
87  }
88 
89  return ret_val;
90 }
