164 void display7SegmentTestPins( gpioMap_t* display7SegmentPins, gpioMap_t pin ){
165 
166  uint8_t i = 0;
167 
168  for(i=0;i<=7;i++){
169  gpioWrite( display7SegmentPins[i], ON );
170  if( i == 0 )
171  gpioWrite( pin, ON );
172  delay(1000);
173  gpioWrite( display7SegmentPins[i], OFF );
174  if( i == 0 )
175  gpioWrite( pin, OFF );
176  }
177 
178 }
